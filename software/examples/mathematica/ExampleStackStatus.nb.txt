Needs["NETLink`"]
LoadNETAssembly["Tinkerforge",NotebookDirectory[]<>"../../.."]

host="localhost"
port=4223
uid="68WRjG"(* Change to your UID *)

(* Create IPConnection and device object *)
ipcon=NETNew["Tinkerforge.IPConnection"]
m=NETNew["Tinkerforge.BrickMaster",uid,ipcon]
ipcon@Connect[host,port]

(* Get voltage and current from stack (in mV/mA) *)
Print["Stack Voltage: "<>ToString[N[Quantity[m@GetStackVoltage[],"mV"]]]]
Print["Stack Current: "<>ToString[N[Quantity[m@GetStackCurrent[],"mA"]]]]

(* Clean up *)
ipcon@Disconnect[]
ReleaseNETObject[m]
ReleaseNETObject[ipcon]
